Will man die Scene realistisch proportional darstellen, nimmt man eine Frustum-Matrix.
Dies hat den Einfluss, das Objekte kleiner erscheinen, je weiter die Scene von einem weg ist.
In der Realität ist dies auch so, das Objekte kleiner erscheinen, je weiter sie von einem weg sind.

procedure TForm1.CreateScene;
const
  w = 1.0;
begin
  Matrix := TMatrix.Create;
  FrustumMatrix := TMatrix.Create;

  FrustumMatrix.Frustum(-w, w, -w, w, 2.5, 1000.0);

  //   FrustumMatrix.Perspective(45, 1.0, 2.5, 1000.0); // Alternativ

  WorldMatrix := TMatrix.Create;
  WorldMatrix.Translate(0, 0, -200.0); // Die Scene in den sichtbaren Bereich verschieben.
  WorldMatrix.Scale(5.0);              // Und der Grösse anpassen.

  WorldMatrix.RotateA(1.5);  // Drehe um Y-Achse
  WorldMatrix.RotateC(pi);  // Drehe um Y-Achse

Vertex-Shader:
#version 330

layout (location = 0) in vec3 inPos; // Vertex-Koordinaten
layout (location = 10) in vec2 inUV; // Textur-Koordinaten

uniform mat4 Matrix;                  // Matrix für die Drehbewegung und Frustum.

out vec2 UV0;

void main(void)
{
  gl_Position = Matrix * vec4(inPos, 1.0);
  UV0 = inUV;
}

Fragment-Shader
#version 330

in vec2 UV0;

uniform sampler2D Sampler;

out vec4 outColor;

void main()
{
//  outColor=vec4(1.0);
  outColor = texture( Sampler, UV0, -5 );
}
